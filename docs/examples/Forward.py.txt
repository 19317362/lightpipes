#PyForward.py
import LightPipes
import matplotlib.pyplot as plt

m=1
nm=1e-9*m
um=1e-6*m
mm=1e-3*m
cm=1e-2*m

size=10*mm
N=64
wavelength=500*nm
R=2*mm
z=1*m
new_size=size
new_N=N

LP=LightPipes.Init()
F=LP.Begin(size,wavelength,N)
F=LP.CircAperture(R,0,0,F)
F=LP.Forward(z,new_size,new_N,F)

I=LP.Intensity(2,F)
plt.imshow(I);plt.axis('off')
plt.show()
